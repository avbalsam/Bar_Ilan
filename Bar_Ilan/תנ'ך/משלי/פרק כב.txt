
 משלי פרק כב (א) נִבְחָר שֵׁם מֵעֹשֶׁר רָב מִכֶּסֶף וּמִזָּהָב חֵן טוֹב: (ב) עָשִׁיר וָרָשׁ נִפְגָּשׁוּ עֹשֵׂה כֻלָּם יְקֹוָק: (ג) עָרוּם רָאָה רָעָה ויסתר ויסתר וְנִסְתָּר וְנִסְתָּר וּפְתָיִים עָבְרוּ וְנֶעֱנָשׁוּ: (ד) עֵקֶב עֲנָוָה יִרְאַת יְקֹוָק עֹשֶׁר וְכָבוֹד וְחַיִּים: (ה) צִנִּים פַּחִים בְּדֶרֶךְ עִקֵּשׁ שׁוֹמֵר נַפְשׁוֹ יִרְחַק מֵהֶם: (ו) חֲנֹךְ לַנַּעַר עַל פִּי דַרְכּוֹ גַּם כִּי יַזְקִין לֹא יָסוּר מִמֶּנָּה: (ז) עָשִׁיר בְּרָשִׁים יִמְשׁוֹל וְעֶבֶד לֹוֶה לְאִישׁ מַלְוֶה: (ח) זוֹרֵעַ עַוְלָה יקצור יקצור יִקְצָר יִקְצָר אָוֶן וְשֵׁבֶט עֶבְרָתוֹ יִכְלֶה: (ט) טוֹב עַיִן הוּא יְבֹרָךְ כִּי נָתַן מִלַּחְמוֹ לַדָּל: (י) גָּרֵשׁ לֵץ וְיֵצֵא מָדוֹן וְיִשְׁבֹּת דִּין וְקָלוֹן: (יא) אֹהֵב טהור טהור טְהָר טְהָר לֵב חֵן שְׂפָתָיו רֵעֵהוּ מֶלֶךְ: (יב) עֵינֵי יְקֹוָק נָצְרוּ דָעַת וַיְסַלֵּף דִּבְרֵי בֹגֵד: (יג) אָמַר עָצֵל אֲרִי בַחוּץ בְּתוֹךְ רְחֹבוֹת אֵרָצֵחַ: (יד) שׁוּחָה עֲמֻקָּה פִּי זָרוֹת זְעוּם יְקֹוָק יפול יפול יִפָּל יִפָּל שָׁם: (טו) אִוֶּלֶת קְשׁוּרָה בְלֶב נָעַר שֵׁבֶט מוּסָר יַרְחִיקֶנָּה מִמֶּנּוּ: (טז) עֹשֵׁק דָּל לְהַרְבּוֹת לוֹ נֹתֵן לְעָשִׁיר אַךְ לְמַחְסוֹר: (יז) הַט אָזְנְךָ וּשְׁמַע דִּבְרֵי חֲכָמִים וְלִבְּךָ תָּשִׁית לְדַעְתִּי: (יח) כִּי נָעִים כִּי תִשְׁמְרֵם בְּבִטְנֶךָ יִכֹּנוּ יַחְדָּו עַל שְׂפָתֶיךָ: (יט) לִהְיוֹת בַּיקֹוָק מִבְטַחֶךָ הוֹדַעְתִּיךָ הַיּוֹם אַף אָתָּה: (כ) הֲלֹא כָתַבְתִּי לְךָ שלשום שלשום שָׁלִישִׁים שָׁלִישִׁים בְּמוֹעֵצֹת וָדָעַת: (כא) לְהוֹדִיעֲךָ קֹשְׁטְ אִמְרֵי אֱמֶת לְהָשִׁיב אֲמָרִים אֱמֶת לְשֹׁלְחֶיךָ: פ (כב) אַל תִּגְזָל דָּל כִּי דַל הוּא וְאַל תְּדַכֵּא עָנִי בַשָּׁעַר: (כג) כִּי יְקֹוָק יָרִיב רִיבָם וְקָבַע אֶת קֹבְעֵיהֶם נָפֶשׁ: (כד) אַל תִּתְרַע אֶת בַּעַל אָף וְאֶת אִישׁ חֵמוֹת לֹא תָבוֹא: (כה) פֶּן תֶּאֱלַף ארחתו ארחתו אֹרְחֹתָיו אֹרְחֹתָיו וְלָקַחְתָּ מוֹקֵשׁ לְנַפְשֶׁךָ: (כו) אַל תְּהִי בְתֹקְעֵי כָף בַּעֹרְבִים מַשָּׁאוֹת: (כז) אִם אֵין לְךָ לְשַׁלֵּם לָמָּה יִקַּח מִשְׁכָּבְךָ מִתַּחְתֶּיךָ: (כח) אַל תַּסֵּג גְּבוּל עוֹלָם אֲשֶׁר עָשׂוּ אֲבוֹתֶיךָ: (כט) חָזִיתָ אִישׁ מָהִיר
