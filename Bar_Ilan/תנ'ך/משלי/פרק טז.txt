
 משלי פרק טז (א) לְאָדָם מַעַרְכֵי לֵב וּמֵיְקֹוָק מַעֲנֵה לָשׁוֹן: (ב) כָּל דַּרְכֵי אִישׁ זַךְ בְּעֵינָיו וְתֹכֵן רוּחוֹת יְקֹוָק: (ג) גֹּל אֶל יְקֹוָק מַעֲשֶׂיךָ וְיִכֹּנוּ מַחְשְׁבֹתֶיךָ: (ד) כֹּל פָּעַל יְקֹוָק לַמַּעֲנֵהוּ וְגַם רָשָׁע לְיוֹם רָעָה: (ה) תּוֹעֲבַת יְקֹוָק כָּל גְּבַהּ לֵב יָד לְיָד לֹא יִנָּקֶה: (ו) בְּחֶסֶד וֶאֱמֶת יְכֻפַּר עָוֹן וּבְיִרְאַת יְקֹוָק סוּר מֵרָע: (ז) בִּרְצוֹת יְקֹוָק דַּרְכֵי אִישׁ גַּם אוֹיְבָיו יַשְׁלִם אִתּוֹ: (ח) טוֹב מְעַט בִּצְדָקָה מֵרֹב תְּבוּאוֹת בְּלֹא מִשְׁפָּט: (ט) לֵב אָדָם יְחַשֵּׁב דַּרְכּוֹ וַיקֹוָק יָכִין צַעֲדוֹ: (י) קֶסֶם עַל שִׂפְתֵי מֶלֶךְ בְּמִשְׁפָּט לֹא יִמְעַל פִּיו: (יא) פֶּלֶס וּמֹאזְנֵי מִשְׁפָּט לַיקֹוָק מַעֲשֵׂהוּ כָּל אַבְנֵי כִיס: (יב) תּוֹעֲבַת מְלָכִים עֲשׂוֹת רֶשַׁע כִּי בִצְדָקָה יִכּוֹן כִּסֵּא: (יג) רְצוֹן מְלָכִים שִׂפְתֵי צֶדֶק וְדֹבֵר יְשָׁרִים יֶאֱהָב: (יד) חֲמַת מֶלֶךְ מַלְאֲכֵי מָוֶת וְאִישׁ חָכָם יְכַפְּרֶנָּה: (טו) בְּאוֹר פְּנֵי מֶלֶךְ חַיִּים וּרְצוֹנוֹ כְּעָב מַלְקוֹשׁ: (טז) קְנֹה חָכְמָה מַה טּוֹב מֵחָרוּץ וּקְנוֹת בִּינָה נִבְחָר מִכָּסֶף: (יז) מְסִלַּת יְשָׁרִים סוּר מֵרָע שֹׁמֵר נַפְשׁוֹ נֹצֵר דַּרְכּוֹ: (יח) לִפְנֵי שֶׁבֶר גָּאוֹן וְלִפְנֵי כִשָּׁלוֹן גֹּבַהּ רוּחַ: (יט) טוֹב שְׁפַל רוּחַ אֶת עניים עניים עֲנָוִים עֲנָוִים מֵחַלֵּק שָׁלָל אֶת גֵּאִים: (כ) מַשְׂכִּיל עַל דָּבָר יִמְצָא טוֹב וּבוֹטֵחַ בַּיקֹוָק אַשְׁרָיו: (כא) לַחֲכַם לֵב יִקָּרֵא נָבוֹן וּמֶתֶק שְׂפָתַיִם יֹסִיף לֶקַח: (כב) מְקוֹר חַיִּים שֵׂכֶל בְּעָלָיו וּמוּסַר אֱוִלִים אִוֶּלֶת: (כג) לֵב חָכָם יַשְׂכִּיל פִּיהוּ וְעַל שְׂפָתָיו יֹסִיף לֶקַח: (כד) צוּף דְּבַשׁ אִמְרֵי נֹעַם מָתוֹק לַנֶּפֶשׁ וּמַרְפֵּא לָעָצֶם: (כה) יֵשׁ דֶּרֶךְ יָשָׁר לִפְנֵי אִישׁ וְאַחֲרִיתָהּ דַּרְכֵי מָוֶת: (כו) נֶפֶשׁ עָמֵל עָמְלָה לּוֹ כִּי אָכַף עָלָיו פִּיהוּ: (כז) אִישׁ בְּלִיַּעַל כֹּרֶה רָעָה וְעַל שפתיו שפתיו שְׂפָתוֹ שְׂפָתוֹ כְּאֵשׁ צָרָבֶת: (כח) אִישׁ תַּהְפֻּכוֹת יְשַׁלַּח מָדוֹן וְנִרְגָּן מַפְרִיד אַלּוּף: (כט) אִישׁ חָמָס יְפַתֶּה רֵעֵהוּ וְהוֹלִיכוֹ בְּדֶרֶךְ לֹא טוֹב: (ל) עֹצֶה עֵינָיו לַחְשֹׁב תַּהְפֻּכוֹת קֹרֵץ שְׂפָתָיו כִּלָּה רָעָה: (לא) עֲטֶרֶת תִּפְאֶרֶת שֵׂיבָה בְּדֶרֶךְ צְדָקָה תִּמָּצֵא: (לב) טוֹב אֶרֶךְ אַפַּיִם מִגִּבּוֹר וּמֹשֵׁל בְּרוּחוֹ מִלֹּכֵד עִיר: (לג) בַּחֵיק יוּטַל אֶת הַגּוֹרָל וּמֵיְקֹוָק כָּל מִשְׁפָּטוֹ:
