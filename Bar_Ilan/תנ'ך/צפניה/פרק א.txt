
 צפניה פרק א (א) דְּבַר יְקֹוָק אֲשֶׁר הָיָה אֶל צְפַנְיָה בֶּן כּוּשִׁי בֶן גְּדַלְיָה בֶּן אֲמַרְיָה בֶּן חִזְקִיָּה בִּימֵי יֹאשִׁיָּהוּ בֶן אָמוֹן מֶלֶךְ יְהוּדָה: (ב) אָסֹף אָסֵף כֹּל מֵעַל פְּנֵי הָאֲדָמָה נְאֻם יְקֹוָק: (ג) אָסֵף אָדָם וּבְהֵמָה אָסֵף עוֹף הַשָּׁמַיִם וּדְגֵי הַיָּם וְהַמַּכְשֵׁלוֹת אֶת הָרְשָׁעִים וְהִכְרַתִּי אֶת הָאָדָם מֵעַל פְּנֵי הָאֲדָמָה נְאֻם יְקֹוָק: (ד) וְנָטִיתִי יָדִי עַל יְהוּדָה וְעַל כָּל יוֹשְׁבֵי יְרוּשָׁלִָם וְהִכְרַתִּי מִן הַמָּקוֹם הַזֶּה אֶת שְׁאָר הַבַּעַל אֶת שֵׁם הַכְּמָרִים עִם הַכֹּהֲנִים: (ה) וְאֶת הַמִּשְׁתַּחֲוִים עַל הַגַּגּוֹת לִצְבָא הַשָּׁמָיִם וְאֶת הַמִּשְׁתַּחֲוִים הַנִּשְׁבָּעִים לַיקֹוָק וְהַנִּשְׁבָּעִים בְּמַלְכָּם: (ו) וְאֶת הַנְּסוֹגִים מֵאַחֲרֵי יְקֹוָק וַאֲשֶׁר לֹא בִקְשׁוּ אֶת יְקֹוָק וְלֹא דְרָשֻׁהוּ: (ז) הַס מִפְּנֵי אֲדֹנָי יְקֹוִק כִּי קָרוֹב יוֹם יְקֹוָק כִּי הֵכִין יְקֹוָק זֶבַח הִקְדִּישׁ קְרֻאָיו: (ח) וְהָיָה בְּיוֹם זֶבַח יְקֹוָק וּפָקַדְתִּי עַל הַשָּׂרִים וְעַל בְּנֵי הַמֶּלֶךְ וְעַל כָּל הַלֹּבְשִׁים מַלְבּוּשׁ נָכְרִי: (ט) וּפָקַדְתִּי עַל כָּל הַדּוֹלֵג עַל הַמִּפְתָּן בַּיּוֹם הַהוּא הַמְמַלְאִים בֵּית אֲדֹנֵיהֶם חָמָס וּמִרְמָה: ס (י) וְהָיָה בַיּוֹם הַהוּא נְאֻם יְקֹוָק קוֹל צְעָקָה מִשַּׁעַר הַדָּגִים וִילָלָה מִן הַמִּשְׁנֶה וְשֶׁבֶר גָּדוֹל מֵהַגְּבָעוֹת: (יא) הֵילִילוּ יֹשְׁבֵי הַמַּכְתֵּשׁ כִּי נִדְמָה כָּל עַם כְּנַעַן נִכְרְתוּ כָּל נְטִילֵי כָסֶף: (יב) וְהָיָה בָּעֵת הַהִיא אֲחַפֵּשׂ אֶת יְרוּשָׁלִַם בַּנֵּרוֹת וּפָקַדְתִּי עַל הָאֲנָשִׁים הַקֹּפְאִים עַל שִׁמְרֵיהֶם הָאֹמְרִים בִּלְבָבָם לֹא יֵיטִיב יְקֹוָק וְלֹא יָרֵעַ: (יג) וְהָיָה חֵילָם לִמְשִׁסָּה וּבָתֵּיהֶם לִשְׁמָמָה וּבָנוּ בָתִּים וְלֹא יֵשֵׁבוּ וְנָטְעוּ כְרָמִים וְלֹא יִשְׁתּוּ אֶת יֵינָם: (יד) קָרוֹב יוֹם יְקֹוָק הַגָּדוֹל קָרוֹב וּמַהֵר מְאֹד קוֹל יוֹם יְקֹוָק מַר צֹרֵחַ שָׁם גִּבּוֹר: (טו) יוֹם עֶבְרָה הַיּוֹם הַהוּא יוֹם צָרָה וּמְצוּקָה יוֹם שֹׁאָה וּמְשׁוֹאָה יוֹם חֹשֶׁךְ וַאֲפֵלָה יוֹם עָנָן וַעֲרָפֶל: (טז) יוֹם שׁוֹפָר וּתְרוּעָה עַל הֶעָרִים הַבְּצֻרוֹת וְעַל הַפִּנּוֹת הַגְּבֹהוֹת: (יז) וַהֲצֵרֹתִי לָאָדָם וְהָלְכוּ כַּעִוְרִים כִּי לַיקֹוָק חָטָאוּ וְשֻׁפַּךְ דָּמָם כֶּעָפָר וּלְחֻמָם כַּגְּלָלִים: (יח) גַּם כַּסְפָּם גַּם זְהָבָם לֹא יוּכַל לְהַצִּילָם בְּיוֹם עֶבְרַת יְקֹוָק וּבְאֵשׁ קִנְאָתוֹ תֵּאָכֵל כָּל הָאָרֶץ כִּי כָלָה אַךְ נִבְהָלָה יַעֲשֶׂה אֵת כָּל יֹשְׁבֵי הָאָרֶץ: ס
