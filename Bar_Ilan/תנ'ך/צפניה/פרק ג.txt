
 צפניה פרק ג (א) הוֹי מֹרְאָה וְנִגְאָלָה הָעִיר הַיּוֹנָה: (ב) לֹא שָׁמְעָה בְּקוֹל לֹא לָקְחָה מוּסָר בַּיקֹוָק לֹא בָטָחָה אֶל אֱלֹהֶיהָ לֹא קָרֵבָה: (ג) שָׂרֶיהָ בְקִרְבָּהּ אֲרָיוֹת שֹׁאֲגִים שֹׁפְטֶיהָ זְאֵבֵי עֶרֶב לֹא גָרְמוּ לַבֹּקֶר: (ד) נְבִיאֶיהָ פֹּחֲזִים אַנְשֵׁי בֹּגְדוֹת כֹּהֲנֶיהָ חִלְּלוּ קֹדֶשׁ חָמְסוּ תּוֹרָה: (ה) יְקֹוָק צַדִּיק בְּקִרְבָּהּ לֹא יַעֲשֶׂה עַוְלָה בַּבֹּקֶר בַּבֹּקֶר מִשְׁפָּטוֹ יִתֵּן לָאוֹר לֹא נֶעְדָּר וְלֹא יוֹדֵעַ עַוָּל בֹּשֶׁת: (ו) הִכְרַתִּי גוֹיִם נָשַׁמּוּ פִּנּוֹתָם הֶחֱרַבְתִּי חוּצוֹתָם מִבְּלִי עוֹבֵר נִצְדּוּ עָרֵיהֶם מִבְּלִי אִישׁ מֵאֵין יוֹשֵׁב: (ז) אָמַרְתִּי אַךְ תִּירְאִי אוֹתִי תִּקְחִי מוּסָר וְלֹא יִכָּרֵת מְעוֹנָהּ כֹּל אֲשֶׁר פָּקַדְתִּי עָלֶיהָ אָכֵן הִשְׁכִּימוּ הִשְׁחִיתוּ כֹּל עֲלִילוֹתָם: (ח) לָכֵן חַכּוּ לִי נְאֻם יְקֹוָק לְיוֹם קוּמִי לְעַד כִּי מִשְׁפָּטִי לֶאֱסֹף גּוֹיִם לְקָבְצִי מַמְלָכוֹת לִשְׁפֹּךְ עֲלֵיהֶם זַעְמִי כֹּל חֲרוֹן אַפִּי כִּי בְּאֵשׁ קִנְאָתִי תֵּאָכֵל כָּל הָאָרֶץ: (ט) כִּי אָז אֶהְפֹּךְ אֶל עַמִּים שָׂפָה בְרוּרָה לִקְרֹא כֻלָּם בְּשֵׁם יְקֹוָק לְעָבְדוֹ שְׁכֶם אֶחָד: (י) מֵעֵבֶר לְנַהֲרֵי כוּשׁ עֲתָרַי בַּת פּוּצַי יוֹבִלוּן מִנְחָתִי: (יא) בַּיּוֹם הַהוּא לֹא תֵבוֹשִׁי מִכֹּל עֲלִילֹתַיִךְ אֲשֶׁר פָּשַׁעַתְּ בִּי כִּי אָז אָסִיר מִקִּרְבֵּךְ עַלִּיזֵי גַּאֲוָתֵךְ וְלֹא תוֹסִפִי לְגָבְהָה עוֹד בְּהַר קָדְשִׁי: (יב) וְהִשְׁאַרְתִּי בְקִרְבֵּךְ עַם עָנִי וָדָל וְחָסוּ בְּשֵׁם יְקֹוָק: (יג) שְׁאֵרִית יִשְׂרָאֵל לֹא יַעֲשׂוּ עַוְלָה וְלֹא יְדַבְּרוּ כָזָב וְלֹא יִמָּצֵא בְּפִיהֶם לְשׁוֹן תַּרְמִית כִּי הֵמָּה יִרְעוּ וְרָבְצוּ וְאֵין מַחֲרִיד: ס (יד) רָנִּי בַּת צִיּוֹן הָרִיעוּ יִשְׂרָאֵל שִׂמְחִי וְעָלְזִי בְּכָל לֵב בַּת יְרוּשָׁלִָם: (טו) הֵסִיר יְקֹוָק מִשְׁפָּטַיִךְ פִּנָּה אֹיְבֵךְ מֶלֶךְ יִשְׂרָאֵל יְקֹוָק בְּקִרְבֵּךְ לֹא תִירְאִי רָע עוֹד: (טז) בַּיּוֹם הַהוּא יֵאָמֵר לִירוּשָׁלִַם אַל תִּירָאִי צִיּוֹן אַל יִרְפּוּ יָדָיִךְ: (יז) יְקֹוָק אֱלֹהַיִךְ בְּקִרְבֵּךְ גִּבּוֹר יוֹשִׁיעַ יָשִׂישׂ עָלַיִךְ בְּשִׂמְחָה יַחֲרִישׁ בְּאַהֲבָתוֹ יָגִיל עָלַיִךְ בְּרִנָּה: (יח) נוּגֵי מִמּוֹעֵד אָסַפְתִּי מִמֵּךְ הָיוּ מַשְׂאֵת עָלֶיהָ חֶרְפָּה: (יט) הִנְנִי עֹשֶׂה אֶת כָּל מְעַנַּיִךְ בָּעֵת הַהִיא וְהוֹשַׁעְתִּי אֶת הַצֹּלֵעָה וְהַנִּדָּחָה אֲקַבֵּץ וְשַׂמְתִּים לִתְהִלָּה וּלְשֵׁם בְּכָל הָאָרֶץ בָּשְׁתָּם: (כ) בָּעֵת הַהִיא אָבִיא אֶתְכֶם וּבָעֵת קַבְּצִי אֶתְכֶם כִּי אֶתֵּן אֶתְכֶם לְשֵׁם וְלִתְהִלָּה בְּכֹל עַמֵּי הָאָרֶץ בְּשׁוּבִי אֶת שְׁבוּתֵיכֶם לְעֵינֵיכֶם אָמַר יְקֹוָק:
