
 ירמיהו פרק מח (א) לְמוֹאָב כֹּה אָמַר יְקֹוָק צְבָאוֹת אֱלֹהֵי יִשְׂרָאֵל הוֹי אֶל נְבוֹ כִּי שֻׁדָּדָה הֹבִישָׁה נִלְכְּדָה קִרְיָתָיִם הֹבִישָׁה הַמִּשְׂגָּב וָחָתָּה: (ב) אֵין עוֹד תְּהִלַּת מוֹאָב בְּחֶשְׁבּוֹן חָשְׁבוּ עָלֶיהָ רָעָה לְכוּ וְנַכְרִיתֶנָּה מִגּוֹי גַּם מַדְמֵן תִּדֹּמִּי אַחֲרַיִךְ תֵּלֶךְ חָרֶב: (ג) קוֹל צְעָקָה מֵחֹרוֹנָיִם שֹׁד וָשֶׁבֶר גָּדוֹל: (ד) נִשְׁבְּרָה מוֹאָב הִשְׁמִיעוּ זְּעָקָה צעוריה צעוריה צְעִירֶיהָ צְעִירֶיהָ : (ה) כִּי מַעֲלֵה הלחות הלחות הַלּוּחִית הַלּוּחִית בִּבְכִי יַעֲלֶה בֶּכִי כִּי בְּמוֹרַד חוֹרֹנַיִם צָרֵי צַעֲקַת שֶׁבֶר שָׁמֵעוּ: (ו) נֻסוּ מַלְּטוּ נַפְשְׁכֶם וְתִהְיֶינָה כַּעֲרוֹעֵר בַּמִּדְבָּר: (ז) כִּי יַעַן בִּטְחֵךְ בְּמַעֲשַׂיִךְ וּבְאוֹצְרוֹתַיִךְ גַּם אַתְּ תִּלָּכֵדִי וְיָצָא כמיש כמיש כְמוֹשׁ כְמוֹשׁ בַּגּוֹלָה כֹּהֲנָיו וְשָׂרָיו יחד יחד יַחְדָּיו יַחְדָּיו : (ח) וְיָבֹא שֹׁדֵד אֶל כָּל עִיר וְעִיר לֹא תִמָּלֵט וְאָבַד הָעֵמֶק וְנִשְׁמַד הַמִּישֹׁר אֲשֶׁר אָמַר יְקֹוָק: (ט) תְּנוּ צִיץ לְמוֹאָב כִּי נָצֹא תֵּצֵא וְעָרֶיהָ לְשַׁמָּה תִהְיֶינָה מֵאֵין יוֹשֵׁב בָּהֵן: (י) אָרוּר עֹשֶׂה מְלֶאכֶת יְקֹוָק רְמִיָּה וְאָרוּר מֹנֵעַ חַרְבּוֹ מִדָּם: (יא) שַׁאֲנַן מוֹאָב מִנְּעוּרָיו וְשֹׁקֵט הוּא אֶל שְׁמָרָיו וְלֹא הוּרַק מִכְּלִי אֶל כֶּלִי וּבַגּוֹלָה לֹא הָלָךְ עַל כֵּן עָמַד טַעְמוֹ בּוֹ וְרֵיחוֹ לֹא נָמָר: ס (יב) לָכֵן הִנֵּה יָמִים בָּאִים נְאֻם יְקֹוָק וְשִׁלַּחְתִּי לוֹ צֹעִים וְצֵעֻהוּ וְכֵלָיו יָרִיקוּ וְנִבְלֵיהֶם יְנַפֵּצוּ: (יג) וּבֹשׁ מוֹאָב מִכְּמוֹשׁ כַּאֲשֶׁר בֹּשׁוּ בֵּית יִשְׂרָאֵל מִבֵּית אֵל מִבְטֶחָם: (יד) אֵיךְ תֹּאמְרוּ גִּבּוֹרִים אֲנָחְנוּ וְאַנְשֵׁי חַיִל לַמִּלְחָמָה: (טו) שֻׁדַּד מוֹאָב וְעָרֶיהָ עָלָה וּמִבְחַר בַּחוּרָיו יָרְדוּ לַטָּבַח נְאֻם הַמֶּלֶךְ יְקֹוָק צְבָאוֹת שְׁמוֹ: (טז) קָרוֹב אֵיד מוֹאָב לָבוֹא וְרָעָתוֹ מִהֲרָה מְאֹד: (יז) נֻדוּ לוֹ כָּל סְבִיבָיו וְכֹל יֹדְעֵי שְׁמוֹ אִמְרוּ אֵיכָה נִשְׁבַּר מַטֵּה עֹז מַקֵּל תִּפְאָרָה: (יח) רְדִי מִכָּבוֹד ישבי ישבי וּשְׁבִי וּשְׁבִי בַצָּמָא יֹשֶׁבֶת בַּת דִּיבוֹן כִּי שֹׁדֵד מוֹאָב עָלָה בָךְ שִׁחֵת מִבְצָרָיִךְ: (יט) אֶל דֶּרֶךְ עִמְדִי וְצַפִּי יוֹשֶׁבֶת עֲרוֹעֵר שַׁאֲלִי נָס וְנִמְלָטָה אִמְרִי מַה נִּהְיָתָה: (כ) הֹבִישׁ מוֹאָב כִּי חַתָּה הילילי הילילי הֵילִילוּ הֵילִילוּ וזעקי וזעקי וּזְעָקוּ וּזְעָקוּ הַגִּידוּ בְאַרְנוֹן כִּי שֻׁדַּד מוֹאָב: (כא) וּמִשְׁפָּט בָּא אֶל אֶרֶץ הַמִּישֹׁר אֶל חֹלוֹן וְאֶל יַהְצָה וְעַל מופעת מופעת מֵיפָעַת מֵיפָעַת : (כב) וְעַל דִּיבוֹן וְעַל נְבוֹ וְעַל בֵּית דִּבְלָתָיִם: (כג) וְעַל קִרְיָתַיִם וְעַל בֵּית גָּמוּל וְעַל בֵּית מְעוֹן: (כד) וְעַל קְרִיּוֹת וְעַל בָּצְרָה וְעַל כָּל עָרֵי אֶרֶץ מוֹאָב הָרְחֹקוֹת וְהַקְּרֹבוֹת: (כה) נִגְדְּעָה קֶרֶן מוֹאָב וּזְרֹעוֹ נִשְׁבָּרָה נְאֻם יְקֹוָק: (כו) הַשְׁכִּירֻהוּ כִּי עַל יְקֹוָק הִגְדִּיל וְסָפַק מוֹאָב בְּקִיאוֹ וְהָיָה לִשְׂחֹק גַּם הוּא: (כז) וְאִם לוֹא הַשְּׂחֹק הָיָה לְךָ יִשְׂרָאֵל אִם בְּגַנָּבִים נמצאה נמצאה נִמְצָא נִמְצָא כִּי מִדֵּי דְבָרֶיךָ בּוֹ תִּתְנוֹדָד: (כח) עִזְבוּ עָרִים וְשִׁכְנוּ בַסֶּלַע יֹשְׁבֵי מוֹאָב וִהְיוּ כְיוֹנָה תְּקַנֵּן בְּעֶבְרֵי פִי פָחַת: (כט) שָׁמַעְנוּ גְאוֹן מוֹאָב גֵּאֶה מְאֹד גָּבְהוֹ וּגְאוֹנוֹ וְגַאֲוָתוֹ וְרֻם לִבּוֹ: (ל) אֲנִי יָדַעְתִּי נְאֻם יְקֹוָק עֶבְרָתוֹ וְלֹא כֵן בַּדָּיו לֹא כֵן עָשׂוּ: (לא) עַל כֵּן עַל מוֹאָב אֲיֵלִיל וּלְמוֹאָב כֻּלֹּה אֶזְעָק אֶל אַנְשֵׁי קִיר חֶרֶשׂ יֶהְגֶּה: (לב) מִבְּכִי יַעְזֵר אֶבְכֶּה לָּךְ הַגֶּפֶן שִׂבְמָה נְטִישֹׁתַיִךְ עָבְרוּ יָם עַד יָם יַעְזֵר נָגָעוּ עַל קֵיצֵךְ וְעַל בְּצִירֵךְ שֹׁדֵד נָפָל: (לג) וְנֶאֶסְפָה שִׂמְחָה וָגִיל מִכַּרְמֶל וּמֵאֶרֶץ מוֹאָב וְיַיִן מִיקָבִים הִשְׁבַּתִּי לֹא יִדְרֹךְ הֵידָד הֵידָד לֹא הֵידָד: (לד) מִזַּעֲקַת חֶשְׁבּוֹן עַד אֶלְעָלֵה עַד יַהַץ נָתְנוּ קוֹלָם מִצֹּעַר עַד חֹרֹנַיִם עֶגְלַת שְׁלִשִׁיָּה כִּי גַּם מֵי נִמְרִים לִמְשַׁמּוֹת יִהְיוּ: (לה) וְהִשְׁבַּתִּי לְמוֹאָב נְאֻם יְקֹוָק מַעֲלֶה בָמָה וּמַקְטִיר לֵאלֹהָיו: (לו) עַל כֵּן לִבִּי לְמוֹאָב כַּחֲלִלִים יֶהֱמֶה וְלִבִּי אֶל אַנְשֵׁי קִיר חֶרֶשׂ כַּחֲלִילִים יֶהֱמֶה עַל כֵּן יִתְרַת עָשָׂה אָבָדוּ: (לז) כִּי כָל רֹאשׁ קָרְחָה וְכָל זָקָן גְּרֻעָה עַל כָּל יָדַיִם גְּדֻדֹת וְעַל מָתְנַיִם שָׂק: (לח) עַל כָּל גַּגּוֹת מוֹאָב וּבִרְחֹבֹתֶיהָ כֻּלֹּה מִסְפֵּד כִּי שָׁבַרְתִּי אֶת מוֹאָב כִּכְלִי אֵין חֵפֶץ בּוֹ נְאֻם יְקֹוָק: (לט) אֵיךְ חַתָּה הֵילִילוּ אֵיךְ הִפְנָה עֹרֶף מוֹאָב בּוֹשׁ וְהָיָה מוֹאָב לִשְׂחֹק וְלִמְחִתָּה לְכָל סְבִיבָיו: ס (מ) כִּי כֹה אָמַר יְקֹוָק הִנֵּה כַנֶּשֶׁר יִדְאֶה וּפָרַשׂ כְּנָפָיו אֶל מוֹאָב: (מא) נִלְכְּדָה הַקְּרִיּוֹת וְהַמְּצָדוֹת נִתְפָּשָׂה וְהָיָה לֵב גִּבּוֹרֵי מוֹאָב בַּיּוֹם הַהוּא כְּלֵב אִשָּׁה מְצֵרָה: (מב) וְנִשְׁמַד מוֹאָב מֵעָם כִּי עַל יְקֹוָק הִגְדִּיל: (מג) פַּחַד וָפַחַת וָפָח עָלֶיךָ יוֹשֵׁב מוֹאָב נְאֻם יְקֹוָק: (מד) הניס הניס הַנָּס הַנָּס מִפְּנֵי הַפַּחַד יִפֹּל אֶל הַפַּחַת וְהָעֹלֶה מִן הַפַּחַת יִלָּכֵד בַּפָּח כִּי אָבִיא אֵלֶיהָ אֶל מוֹאָב שְׁנַת פְּקֻדָּתָם נְאֻם יְקֹוָק: (מה) בְּצֵל חֶשְׁבּוֹן עָמְדוּ מִכֹּחַ נָסִים כִּי אֵשׁ יָצָא מֵחֶשְׁבּוֹן וְלֶהָבָה מִבֵּין סִיחוֹן וַתֹּאכַל פְּאַת מוֹאָב וְקָדְקֹד בְּנֵי שָׁאוֹן: (מו) אוֹי לְךָ מוֹאָב אָבַד עַם כְּמוֹשׁ כִּי לֻקְּחוּ בָנֶיךָ בַּשֶּׁבִי וּבְנֹתֶיךָ בַּשִּׁבְיָה: (מז) וְשַׁבְתִּי שְׁבוּת מוֹאָב בְּאַחֲרִית הַיָּמִים נְאֻם יְקֹוָק עַד הֵנָּה מִשְׁפַּט מוֹאָב: ס
