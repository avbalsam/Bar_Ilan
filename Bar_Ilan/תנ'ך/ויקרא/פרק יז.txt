
 ויקרא פרשת אחרי מות פרק יז (א) וַיְדַבֵּר יְקֹוָק אֶל מֹשֶׁה לֵּאמֹר: (ב) דַּבֵּר אֶל אַהֲרֹן וְאֶל בָּנָיו וְאֶל כָּל בְּנֵי יִשְׂרָאֵל וְאָמַרְתָּ אֲלֵיהֶם זֶה הַדָּבָר אֲשֶׁר צִוָּה יְקֹוָק לֵאמֹר: (ג) אִישׁ אִישׁ מִבֵּית יִשְׂרָאֵל אֲשֶׁר יִשְׁחַט שׁוֹר אוֹ כֶשֶׂב אוֹ עֵז בַּמַּחֲנֶה אוֹ אֲשֶׁר יִשְׁחַט מִחוּץ לַמַּחֲנֶה: (ד) וְאֶל פֶּתַח אֹהֶל מוֹעֵד לֹא הֱבִיאוֹ לְהַקְרִיב קָרְבָּן לַיקֹוָק לִפְנֵי מִשְׁכַּן יְקֹוָק דָּם יֵחָשֵׁב לָאִישׁ הַהוּא דָּם שָׁפָךְ וְנִכְרַת הָאִישׁ הַהוּא מִקֶּרֶב עַמּוֹ: (ה) לְמַעַן אֲשֶׁר יָבִיאוּ בְּנֵי יִשְׂרָאֵל אֶת זִבְחֵיהֶם אֲשֶׁר הֵם זֹבְחִים עַל פְּנֵי הַשָּׂדֶה וֶהֱבִיאֻם לַיקֹוָק אֶל פֶּתַח אֹהֶל מוֹעֵד אֶל הַכֹּהֵן וְזָבְחוּ זִבְחֵי שְׁלָמִים לַיקֹוָק אוֹתָם: (ו) וְזָרַק הַכֹּהֵן אֶת הַדָּם עַל מִזְבַּח יְקֹוָק פֶּתַח אֹהֶל מוֹעֵד וְהִקְטִיר הַחֵלֶב לְרֵיחַ נִיחֹחַ לַיקֹוָק: (ז) וְלֹא יִזְבְּחוּ עוֹד אֶת זִבְחֵיהֶם לַשְּׂעִירִם אֲשֶׁר הֵם זֹנִים אַחֲרֵיהֶם חֻקַּת עוֹלָם תִּהְיֶה זֹּאת לָהֶם לְדֹרֹתָם: (ח) וַאֲלֵהֶם תֹּאמַר אִישׁ אִישׁ מִבֵּית יִשְׂרָאֵל וּמִן הַגֵּר אֲשֶׁר יָגוּר בְּתוֹכָם אֲשֶׁר יַעֲלֶה עֹלָה אוֹ זָבַח: (ט) וְאֶל פֶּתַח אֹהֶל מוֹעֵד לֹא יְבִיאֶנּוּ לַעֲשׂוֹת אֹתוֹ לַיקֹוָק וְנִכְרַת הָאִישׁ הַהוּא מֵעַמָּיו: (י) וְאִישׁ אִישׁ מִבֵּית יִשְׂרָאֵל וּמִן הַגֵּר הַגָּר בְּתוֹכָם אֲשֶׁר יֹאכַל כָּל דָּם וְנָתַתִּי פָנַי בַּנֶּפֶשׁ הָאֹכֶלֶת אֶת הַדָּם וְהִכְרַתִּי אֹתָהּ מִקֶּרֶב עַמָּהּ: (יא) כִּי נֶפֶשׁ הַבָּשָׂר בַּדָּם הִוא וַאֲנִי נְתַתִּיו לָכֶם עַל הַמִּזְבֵּחַ לְכַפֵּר עַל נַפְשֹׁתֵיכֶם כִּי הַדָּם הוּא בַּנֶּפֶשׁ יְכַפֵּר: (יב) עַל כֵּן אָמַרְתִּי לִבְנֵי יִשְׂרָאֵל כָּל נֶפֶשׁ מִכֶּם לֹא תֹאכַל דָּם וְהַגֵּר הַגָּר בְּתוֹכְכֶם לֹא יֹאכַל דָּם: (יג) וְאִישׁ אִישׁ מִבְּנֵי יִשְׂרָאֵל וּמִן הַגֵּר הַגָּר בְּתוֹכָם אֲשֶׁר יָצוּד צֵיד חַיָּה אוֹ עוֹף אֲשֶׁר יֵאָכֵל וְשָׁפַךְ אֶת דָּמוֹ וְכִסָּהוּ בֶּעָפָר: (יד) כִּי נֶפֶשׁ כָּל בָּשָׂר דָּמוֹ בְנַפְשׁוֹ הוּא וָאֹמַר לִבְנֵי יִשְׂרָאֵל דַּם כָּל בָּשָׂר לֹא תֹאכֵלוּ כִּי נֶפֶשׁ כָּל בָּשָׂר דָּמוֹ הִוא כָּל אֹכְלָיו יִכָּרֵת: (טו) וְכָל נֶפֶשׁ אֲשֶׁר תֹּאכַל נְבֵלָה וּטְרֵפָה בָּאֶזְרָח וּבַגֵּר וְכִבֶּס בְּגָדָיו וְרָחַץ בַּמַּיִם וְטָמֵא עַד הָעֶרֶב וְטָהֵר: (טז) וְאִם לֹא יְכַבֵּס וּבְשָׂרוֹ לֹא יִרְחָץ וְנָשָׂא עֲוֹנוֹ: פ
