
 תהלים פרק קמז (א) הַלְלוּ יָהּ כִּי טוֹב זַמְּרָה אֱלֹהֵינוּ כִּי נָעִים נָאוָה תְהִלָּה: (ב) בּוֹנֵה יְרוּשָׁלִַם יְקֹוָק נִדְחֵי יִשְׂרָאֵל יְכַנֵּס: (ג) הָרֹפֵא לִשְׁבוּרֵי לֵב וּמְחַבֵּשׁ לְעַצְּבוֹתָם: (ד) מוֹנֶה מִסְפָּר לַכּוֹכָבִים לְכֻלָּם שֵׁמוֹת יִקְרָא: (ה) גָּדוֹל אֲדוֹנֵינוּ וְרַב כֹּחַ לִתְבוּנָתוֹ אֵין מִסְפָּר: (ו) מְעוֹדֵד עֲנָוִים יְקֹוָק מַשְׁפִּיל רְשָׁעִים עֲדֵי אָרֶץ: (ז) עֱנוּ לַיקֹוָק בְּתוֹדָה זַמְּרוּ לֵאלֹהֵינוּ בְכִנּוֹר: (ח) הַמְכַסֶּה שָׁמַיִם בְּעָבִים הַמֵּכִין לָאָרֶץ מָטָר הַמַּצְמִיחַ הָרִים חָצִיר: (ט) נוֹתֵן לִבְהֵמָה לַחְמָהּ לִבְנֵי עֹרֵב אֲשֶׁר יִקְרָאוּ: (י) לֹא בִגְבוּרַת הַסּוּס יֶחְפָּץ לֹא בְשׁוֹקֵי הָאִישׁ יִרְצֶה: (יא) רוֹצֶה יְקֹוָק אֶת יְרֵאָיו אֶת הַמְיַחֲלִים לְחַסְדּוֹ: (יב) שַׁבְּחִי יְרוּשָׁלִַם אֶת יְקֹוָק הַלְלִי אֱלֹהַיִךְ צִיּוֹן: (יג) כִּי חִזַּק בְּרִיחֵי שְׁעָרָיִךְ בֵּרַךְ בָּנַיִךְ בְּקִרְבֵּךְ: (יד) הַשָּׂם גְּבוּלֵךְ שָׁלוֹם חֵלֶב חִטִּים יַשְׂבִּיעֵךְ: (טו) הַשֹּׁלֵחַ אִמְרָתוֹ אָרֶץ עַד מְהֵרָה יָרוּץ דְּבָרוֹ: (טז) הַנֹּתֵן שֶׁלֶג כַּצָּמֶר כְּפוֹר כָּאֵפֶר יְפַזֵּר: (יז) מַשְׁלִיךְ קַרְחוֹ כְפִתִּים לִפְנֵי קָרָתוֹ מִי יַעֲמֹד: (יח) יִשְׁלַח דְּבָרוֹ וְיַמְסֵם יַשֵּׁב רוּחוֹ יִזְּלוּ מָיִם: (יט) מַגִּיד דברו דברו דְּבָרָיו דְּבָרָיו לְיַעֲקֹב חֻקָּיו וּמִשְׁפָּטָיו לְיִשְׂרָאֵל: (כ) לֹא עָשָׂה כֵן לְכָל גּוֹי וּמִשְׁפָּטִים בַּל יְדָעוּם הַלְלוּ יָהּ:
