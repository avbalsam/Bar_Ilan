
 תהלים פרק קיח (א) הוֹדוּ לַיקֹוָק כִּי טוֹב כִּי לְעוֹלָם חַסְדּוֹ: (ב) יֹאמַר נָא יִשְׂרָאֵל כִּי לְעוֹלָם חַסְדּוֹ: (ג) יֹאמְרוּ נָא בֵית אַהֲרֹן כִּי לְעוֹלָם חַסְדּוֹ: (ד) יֹאמְרוּ נָא יִרְאֵי יְקֹוָק כִּי לְעוֹלָם חַסְדּוֹ: (ה) מִן הַמֵּצַר קָרָאתִי יָּהּ עָנָנִי בַמֶּרְחָב יָהּ: (ו) יְקֹוָק לִי לֹא אִירָא מַה יַּעֲשֶׂה לִי אָדָם: (ז) יְקֹוָק לִי בְּעֹזְרָי וַאֲנִי אֶרְאֶה בְשֹׂנְאָי: (ח) טוֹב לַחֲסוֹת בַּיקֹוָק מִבְּטֹחַ בָּאָדָם: (ט) טוֹב לַחֲסוֹת בַּיקֹוָק מִבְּטֹחַ בִּנְדִיבִים: (י) כָּל גּוֹיִם סְבָבוּנִי בְּשֵׁם יְקֹוָק כִּי אֲמִילַם: (יא) סַבּוּנִי גַם סְבָבוּנִי בְּשֵׁם יְקֹוָק כִּי אֲמִילַם: (יב) סַבּוּנִי כִדְבוֹרִים דֹּעֲכוּ כְּאֵשׁ קוֹצִים בְּשֵׁם יְקֹוָק כִּי אֲמִילַם: (יג) דַּחֹה דְחִיתַנִי לִנְפֹּל וַיקֹוָק עֲזָרָנִי: (יד) עָזִּי וְזִמְרָת יָהּ וַיְהִי לִי לִישׁוּעָה: (טו) קוֹל רִנָּה וִישׁוּעָה בְּאָהֳלֵי צַדִּיקִים יְמִין יְקֹוָק עֹשָׂה חָיִל: (טז) יְמִין יְקֹוָק רוֹמֵמָה יְמִין יְקֹוָק עֹשָׂה חָיִל: (יז) לֹא אָמוּת כִּי אֶחְיֶה וַאֲסַפֵּר מַעֲשֵׂי יָהּ: (יח) יַסֹּר יִסְּרַנִּי יָּהּ וְלַמָּוֶת לֹא נְתָנָנִי: (יט) פִּתְחוּ לִי שַׁעֲרֵי צֶדֶק אָבֹא בָם אוֹדֶה יָהּ: (כ) זֶה הַשַּׁעַר לַיקֹוָק צַדִּיקִים יָבֹאוּ בוֹ: (כא) אוֹדְךָ כִּי עֲנִיתָנִי וַתְּהִי לִי לִישׁוּעָה: (כב) אֶבֶן מָאֲסוּ הַבּוֹנִים הָיְתָה לְרֹאשׁ פִּנָּה: (כג) מֵאֵת יְקֹוָק הָיְתָה זֹּאת הִיא נִפְלָאת בְּעֵינֵינוּ: (כד) זֶה הַיּוֹם עָשָׂה יְקֹוָק נָגִילָה וְנִשְׂמְחָה בוֹ: (כה) אָנָּא יְקֹוָק הוֹשִׁיעָה נָּא אָנָּא יְקֹוָק הַצְלִיחָה נָּא: (כו) בָּרוּךְ הַבָּא בְּשֵׁם יְקֹוָק בֵּרַכְנוּכֶם מִבֵּית יְקֹוָק: (כז) אֵל יְקֹוָק וַיָּאֶר לָנוּ אִסְרוּ חַג בַּעֲבֹתִים עַד קַרְנוֹת הַמִּזְבֵּחַ: (כח) אֵלִי אַתָּה וְאוֹדֶךָּ אֱלֹהַי אֲרוֹמְמֶךָּ: (כט) הוֹדוּ לַיקֹוָק כִּי טוֹב כִּי לְעוֹלָם חַסְדּוֹ:
