
 ישעיהו פרק ח (א) וַיֹּאמֶר יְקֹוָק אֵלַי קַח לְךָ גִּלָּיוֹן גָּדוֹל וּכְתֹב עָלָיו בְּחֶרֶט אֱנוֹשׁ לְמַהֵר שָׁלָל חָשׁ בַּז: (ב) וְאָעִידָה לִּי עֵדִים נֶאֱמָנִים אֵת אוּרִיָּה הַכֹּהֵן וְאֶת זְכַרְיָהוּ בֶּן יְבֶרֶכְיָהוּ: (ג) וָאֶקְרַב אֶל הַנְּבִיאָה וַתַּהַר וַתֵּלֶד בֵּן וַיֹּאמֶר יקֹוָק אֵלַי קְרָא שְׁמוֹ מַהֵר שָׁלָל חָשׁ בַּז: (ד) כִּי בְּטֶרֶם יֵדַע הַנַּעַר קְרֹא אָבִי וְאִמִּי יִשָּׂא אֶת חֵיל דַּמֶּשֶׂק וְאֵת שְׁלַל שֹׁמְרוֹן לִפְנֵי מֶלֶךְ אַשּׁוּר: ס (ה) וַיֹּסֶף יְקֹוָק דַּבֵּר אֵלַי עוֹד לֵאמֹר: (ו) יַעַן כִּי מָאַס הָעָם הַזֶּה אֵת מֵי הַשִּׁלֹחַ הַהֹלְכִים לְאַט וּמְשׂוֹשׂ אֶת רְצִין וּבֶן רְמַלְיָהוּ: (ז) וְלָכֵן הִנֵּה אֲדֹנָי מַעֲלֶה עֲלֵיהֶם אֶת מֵי הַנָּהָר הָעֲצוּמִים וְהָרַבִּים אֶת מֶלֶךְ אַשּׁוּר וְאֶת כָּל כְּבוֹדוֹ וְעָלָה עַל כָּל אֲפִיקָיו וְהָלַךְ עַל כָּל גְּדוֹתָיו: (ח) וְחָלַף בִּיהוּדָה שָׁטַף וְעָבַר עַד צַוָּאר יַגִּיעַ וְהָיָה מֻטּוֹת כְּנָפָיו מְלֹא רֹחַב אַרְצְךָ עִמָּנוּ אֵל: ס (ט) רֹעוּ עַמִּים וָחֹתּוּ וְהַאֲזִינוּ כֹּל מֶרְחַקֵּי אָרֶץ הִתְאַזְּרוּ וָחֹתּוּ הִתְאַזְּרוּ וָחֹתּוּ: (י) עֻצוּ עֵצָה וְתֻפָר דַּבְּרוּ דָבָר וְלֹא יָקוּם כִּי עִמָּנוּ אֵל: ס (יא) כִּי כֹה אָמַר יְקֹוָק אֵלַי כְּחֶזְקַת הַיָּד וְיִסְּרֵנִי מִלֶּכֶת בְּדֶרֶךְ הָעָם הַזֶּה לֵאמֹר: (יב) לֹא תֹאמְרוּן קֶשֶׁר לְכֹל אֲשֶׁר יֹאמַר הָעָם הַזֶּה קָשֶׁר וְאֶת מוֹרָאוֹ לֹא תִירְאוּ וְלֹא תַעֲרִיצוּ: (יג) אֶת יְקֹוָק צְבָאוֹת אֹתוֹ תַקְדִּישׁוּ וְהוּא מוֹרַאֲכֶם וְהוּא מַעֲרִצְכֶם: (יד) וְהָיָה לְמִקְדָּשׁ וּלְאֶבֶן נֶגֶף וּלְצוּר מִכְשׁוֹל לִשְׁנֵי בָתֵּי יִשְׂרָאֵל לְפַח וּלְמוֹקֵשׁ לְיוֹשֵׁב יְרוּשָׁלִָם: (טו) וְכָשְׁלוּ בָם רַבִּים וְנָפְלוּ וְנִשְׁבָּרוּ וְנוֹקְשׁוּ וְנִלְכָּדוּ: ס (טז) צוֹר תְּעוּדָה חֲתוֹם תּוֹרָה בְּלִמֻּדָי: (יז) וְחִכִּיתִי לַיקֹוָק הַמַּסְתִּיר פָּנָיו מִבֵּית יַעֲקֹב וְקִוֵּיתִי לוֹ: (יח) הִנֵּה אָנֹכִי וְהַיְלָדִים אֲשֶׁר נָתַן לִי יְקֹוָק לְאֹתוֹת וּלְמוֹפְתִים בְּיִשְׂרָאֵל מֵעִם יְקֹוָק צְבָאוֹת הַשֹּׁכֵן בְּהַר צִיּוֹן: ס (יט) וְכִי יֹאמְרוּ אֲלֵיכֶם דִּרְשׁוּ אֶל הָאֹבוֹת וְאֶל הַיִּדְּעֹנִים הַמְצַפְצְפִים וְהַמַּהְגִּים הֲלוֹא עַם אֶל אֱלֹהָיו יִדְרֹשׁ בְּעַד הַחַיִּים אֶל הַמֵּתִים: (כ) לְתוֹרָה וְלִתְעוּדָה אִם לֹא יֹאמְרוּ כַּדָּבָר הַזֶּה אֲשֶׁר אֵין לוֹ שָׁחַר: (כא) וְעָבַר בָּהּ נִקְשֶׁה וְרָעֵב וְהָיָה כִי יִרְעַב וְהִתְקַצַּף וְקִלֵּל בְּמַלְכּוֹ וּבֵאלֹהָיו וּפָנָה לְמָעְלָה: (כב) וְאֶל אֶרֶץ יַבִּיט וְהִנֵּה צָרָה וַחֲשֵׁכָה מְעוּף צוּקָה וַאֲפֵלָה מְנֻדָּח: (כג) כִּי לֹא מוּעָף לַאֲשֶׁר מוּצָק לָהּ כָּעֵת הָרִאשׁוֹן הֵקַל אַרְצָה זְבֻלוּן וְאַרְצָה נַפְתָּלִי וְהָאַחֲרוֹן הִכְבִּיד דֶּרֶךְ הַיָּם עֵבֶר הַיַּרְדֵּן גְּלִיל הַגּוֹיִם:
