
 ישעיהו פרק נב (א) עוּרִי עוּרִי לִבְשִׁי עֻזֵּךְ צִיּוֹן לִבְשִׁי בִּגְדֵי תִפְאַרְתֵּךְ יְרוּשָׁלִַם עִיר הַקֹּדֶשׁ כִּי לֹא יוֹסִיף יָבֹא בָךְ עוֹד עָרֵל וְטָמֵא: (ב) הִתְנַעֲרִי מֵעָפָר קוּמִי שְּׁבִי יְרוּשָׁלִָם התפתחו התפתחו הִתְפַּתְּחִי הִתְפַּתְּחִי מוֹסְרֵי צַוָּארֵךְ שְׁבִיָּה בַּת צִיּוֹן: ס (ג) כִּי כֹה אָמַר יְקֹוָק חִנָּם נִמְכַּרְתֶּם וְלֹא בְכֶסֶף תִּגָּאֵלוּ: (ד) כִּי כֹה אָמַר אֲדֹנָי יְקֹוִק מִצְרַיִם יָרַד עַמִּי בָרִאשֹׁנָה לָגוּר שָׁם וְאַשּׁוּר בְּאֶפֶס עֲשָׁקוֹ: (ה) וְעַתָּה מי מי מַה מַה לִּי פֹה נְאֻם יְקֹוָק כִּי לֻקַּח עַמִּי חִנָּם משלו משלו מֹשְׁלָיו מֹשְׁלָיו יְהֵילִילוּ נְאֻם יְקֹוָק וְתָמִיד כָּל הַיּוֹם שְׁמִי מִנֹּאָץ: (ו) לָכֵן יֵדַע עַמִּי שְׁמִי לָכֵן בַּיּוֹם הַהוּא כִּי אֲנִי הוּא הַמְדַבֵּר הִנֵּנִי: (ז) מַה נָּאווּ עַל הֶהָרִים רַגְלֵי מְבַשֵּׂר מַשְׁמִיעַ שָׁלוֹם מְבַשֵּׂר טוֹב מַשְׁמִיעַ יְשׁוּעָה אֹמֵר לְצִיּוֹן מָלַךְ אֱלֹהָיִךְ: (ח) קוֹל צֹפַיִךְ נָשְׂאוּ קוֹל יַחְדָּו יְרַנֵּנוּ כִּי עַיִן בְּעַיִן יִרְאוּ בְּשׁוּב יְקֹוָק צִיּוֹן: (ט) פִּצְחוּ רַנְּנוּ יַחְדָּו חָרְבוֹת יְרוּשָׁלִָם כִּי נִחַם יְקֹוָק עַמּוֹ גָּאַל יְרוּשָׁלִָם: (י) חָשַׂף יְקֹוָק אֶת זְרוֹעַ קָדְשׁוֹ לְעֵינֵי כָּל הַגּוֹיִם וְרָאוּ כָּל אַפְסֵי אָרֶץ אֵת יְשׁוּעַת אֱלֹהֵינוּ: ס (יא) סוּרוּ סוּרוּ צְאוּ מִשָּׁם טָמֵא אַל תִּגָּעוּ צְאוּ מִתּוֹכָהּ הִבָּרוּ נֹשְׂאֵי כְּלֵי יְקֹוָק: (יב) כִּי לֹא בְחִפָּזוֹן תֵּצֵאוּ וּבִמְנוּסָה לֹא תֵלֵכוּן כִּי הֹלֵךְ לִפְנֵיכֶם יְקֹוָק וּמְאַסִּפְכֶם אֱלֹהֵי יִשְׂרָאֵל: ס (יג) הִנֵּה יַשְׂכִּיל עַבְדִּי יָרוּם וְנִשָּׂא וְגָבַהּ מְאֹד: (יד) כַּאֲשֶׁר שָׁמְמוּ עָלֶיךָ רַבִּים כֵּן מִשְׁחַת מֵאִישׁ מַרְאֵהוּ וְתֹאֲרוֹ מִבְּנֵי אָדָם: (טו) כֵּן יַזֶּה גּוֹיִם רַבִּים עָלָיו יִקְפְּצוּ מְלָכִים פִּיהֶם כִּי אֲשֶׁר לֹא סֻפַּר לָהֶם רָאוּ וַאֲשֶׁר לֹא שָׁמְעוּ הִתְבּוֹנָנוּ:
