
 ישעיהו פרק כד (א) הִנֵּה יְקֹוָק בּוֹקֵק הָאָרֶץ וּבוֹלְקָהּ וְעִוָּה פָנֶיהָ וְהֵפִיץ יֹשְׁבֶיהָ: (ב) וְהָיָה כָעָם כַּכֹּהֵן כַּעֶבֶד כַּאדֹנָיו כַּשִּׁפְחָה כַּגְּבִרְתָּהּ כַּקּוֹנֶה כַּמּוֹכֵר כַּמַּלְוֶה כַּלֹּוֶה כַּנֹּשֶׁה כַּאֲשֶׁר נֹשֶׁא בוֹ: (ג) הִבּוֹק תִּבּוֹק הָאָרֶץ וְהִבּוֹז תִּבּוֹז כִּי יְקֹוָק דִּבֶּר אֶת הַדָּבָר הַזֶּה: (ד) אָבְלָה נָבְלָה הָאָרֶץ אֻמְלְלָה נָבְלָה תֵּבֵל אֻמְלָלוּ מְרוֹם עַם הָאָרֶץ: (ה) וְהָאָרֶץ חָנְפָה תַּחַת יֹשְׁבֶיהָ כִּי עָבְרוּ תוֹרֹת חָלְפוּ חֹק הֵפֵרוּ בְּרִית עוֹלָם: (ו) עַל כֵּן אָלָה אָכְלָה אֶרֶץ וַיֶּאְשְׁמוּ יֹשְׁבֵי בָהּ עַל כֵּן חָרוּ יֹשְׁבֵי אֶרֶץ וְנִשְׁאַר אֱנוֹשׁ מִזְעָר: (ז) אָבַל תִּירוֹשׁ אֻמְלְלָה גָפֶן נֶאֶנְחוּ כָּל שִׂמְחֵי לֵב: (ח) שָׁבַת מְשׂוֹשׂ תֻּפִּים חָדַל שְׁאוֹן עַלִּיזִים שָׁבַת מְשׂוֹשׂ כִּנּוֹר: (ט) בַּשִּׁיר לֹא יִשְׁתּוּ יָיִן יֵמַר שֵׁכָר לְשֹׁתָיו: (י) נִשְׁבְּרָה קִרְיַת תֹּהוּ סֻגַּר כָּל בַּיִת מִבּוֹא: (יא) צְוָחָה עַל הַיַּיִן בַּחוּצוֹת עָרְבָה כָּל שִׂמְחָה גָּלָה מְשׂוֹשׂ הָאָרֶץ: (יב) נִשְׁאַר בָּעִיר שַׁמָּה וּשְׁאִיָּה יֻכַּת שָׁעַר: (יג) כִּי כֹה יִהְיֶה בְּקֶרֶב הָאָרֶץ בְּתוֹךְ הָעַמִּים כְּנֹקֶף זַיִת כְּעוֹלֵלֹת אִם כָּלָה בָצִיר: (יד) הֵמָּה יִשְׂאוּ קוֹלָם יָרֹנּוּ בִּגְאוֹן יְקֹוָק צָהֲלוּ מִיָּם: (טו) עַל כֵּן בָּאֻרִים כַּבְּדוּ יְקֹוָק בְּאִיֵּי הַיָּם שֵׁם יְקֹוָק אֱלֹהֵי יִשְׂרָאֵל: ס (טז) מִכְּנַף הָאָרֶץ זְמִרֹת שָׁמַעְנוּ צְבִי לַצַּדִּיק וָאֹמַר רָזִי לִי רָזִי לִי אוֹי לִי בֹּגְדִים בָּגָדוּ וּבֶגֶד בּוֹגְדִים בָּגָדוּ: (יז) פַּחַד וָפַחַת וָפָח עָלֶיךָ יוֹשֵׁב הָאָרֶץ: (יח) וְהָיָה הַנָּס מִקּוֹל הַפַּחַד יִפֹּל אֶל הַפַּחַת וְהָעוֹלֶה מִתּוֹךְ הַפַּחַת יִלָּכֵד בַּפָּח כִּי אֲרֻבּוֹת מִמָּרוֹם נִפְתָּחוּ וַיִּרְעֲשׁוּ מוֹסְדֵי אָרֶץ: (יט) רֹעָה הִתְרֹעֲעָה הָאָרֶץ פּוֹר הִתְפּוֹרְרָה אֶרֶץ מוֹט הִתְמוֹטְטָה אָרֶץ: (כ) נוֹעַ תָּנוּעַ אֶרֶץ כַּשִּׁכּוֹר וְהִתְנוֹדְדָה כַּמְּלוּנָה וְכָבַד עָלֶיהָ פִּשְׁעָהּ וְנָפְלָה וְלֹא תֹסִיף קוּם: ס (כא) וְהָיָה בַּיּוֹם הַהוּא יִפְקֹד יְקֹוָק עַל צְבָא הַמָּרוֹם בַּמָּרוֹם וְעַל מַלְכֵי הָאֲדָמָה עַל הָאֲדָמָה: (כב) וְאֻסְּפוּ אֲסֵפָה אַסִּיר עַל בּוֹר וְסֻגְּרוּ עַל מַסְגֵּר וּמֵרֹב יָמִים יִפָּקֵדוּ: (כג) וְחָפְרָה הַלְּבָנָה וּבוֹשָׁה הַחַמָּה כִּי מָלַךְ יְקֹוָק צְבָאוֹת בְּהַר צִיּוֹן וּבִירוּשָׁלִַם וְנֶגֶד זְקֵנָיו כָּבוֹד: פ
