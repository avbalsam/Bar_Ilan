
 שמואל ב פרק כג (א) וְאֵלֶּה דִּבְרֵי דָוִד הָאַחֲרֹנִים נְאֻם דָּוִד בֶּן יִשַׁי וּנְאֻם הַגֶּבֶר הֻקַם עָל מְשִׁיחַ אֱלֹהֵי יַעֲקֹב וּנְעִים זְמִרוֹת יִשְׂרָאֵל: (ב) רוּחַ יְקֹוָק דִּבֶּר בִּי וּמִלָּתוֹ עַל לְשׁוֹנִי: (ג) אָמַר אֱלֹהֵי יִשְׂרָאֵל לִי דִבֶּר צוּר יִשְׂרָאֵל מוֹשֵׁל בָּאָדָם צַדִּיק מוֹשֵׁל יִרְאַת אֱלֹהִים: (ד) וּכְאוֹר בֹּקֶר יִזְרַח שָׁמֶשׁ בֹּקֶר לֹא עָבוֹת מִנֹּגַהּ מִמָּטָר דֶּשֶׁא מֵאָרֶץ: (ה) כִּי לֹא כֵן בֵּיתִי עִם אֵל כִּי בְרִית עוֹלָם שָׂם לִי עֲרוּכָה בַכֹּל וּשְׁמֻרָה כִּי כָל יִשְׁעִי וְכָל חֵפֶץ כִּי לֹא יַצְמִיחַ: (ו) וּבְלִיַּעַל כְּקוֹץ מֻנָד כֻּלָּהַם כִּי לֹא בְיָד יִקָּחוּ: (ז) וְאִישׁ יִגַּע בָּהֶם יִמָּלֵא בַרְזֶל וְעֵץ חֲנִית וּבָאֵשׁ שָׂרוֹף יִשָּׂרְפוּ בַּשָּׁבֶת: פ (ח) אֵלֶּה שְׁמוֹת הַגִּבֹּרִים אֲשֶׁר לְדָוִד יֹשֵׁב בַּשֶּׁבֶת תַּחְכְּמֹנִי רֹאשׁ הַשָּׁלִשִׁי הוּא עֲדִינוֹ העצנו העצנו הָעֶצְנִי הָעֶצְנִי עַל שְׁמֹנֶה מֵאוֹת חָלָל בְּפַעַם אחד אחד אֶחָת אֶחָת : ס (ט) ואחרו ואחרו וְאַחֲרָיו וְאַחֲרָיו אֶלְעָזָר בֶּן דדי דדי דֹּדוֹ דֹּדוֹ בֶּן אֲחֹחִי בִּשְׁלֹשָׁה גברים גברים הַגִּבֹּרִים הַגִּבֹּרִים עִם דָּוִד בְּחָרְפָם בַּפְּלִשְׁתִּים נֶאֶסְפוּ שָׁם לַמִּלְחָמָה וַיַּעֲלוּ אִישׁ יִשְׂרָאֵל: (י) הוּא קָם וַיַּךְ אָגֵא הָרָרִי וַיֵּאָסְפוּ פְלִשְׁתִּים לַחַיָּה וַתְּהִי שָׁם חֶלְקַת הַשָּׂדֶה מְלֵאָה עֲדָשִׁים וְהָעָם נָס מִפְּנֵי פְלִשְׁתִּים: (יב) וַיִּתְיַצֵּב בְּתוֹךְ הַחֶלְקָה וַיַּצִּילֶהָ וַיַּךְ אֶת פְּלִשְׁתִּים וַיַּעַשׂ יְקֹוָק תְּשׁוּעָה גְדוֹלָה: ס (יג) וַיֵּרְדוּ שלשים שלשים שְׁלֹשָׁה שְׁלֹשָׁה מֵהַשְּׁלֹשִׁים רֹאשׁ וַיָּבֹאוּ אֶל קָצִיר אֶל דָּוִד אֶל מְעָרַת עֲדֻלָּם וְחַיַּת פְּלִשְׁתִּים חֹנָה בְּעֵמֶק רְפָאִים: (יד) וְדָוִד אָז בַּמְּצוּדָה וּמַצַּב פְּלִשְׁתִּים אָז בֵּית לָחֶם: (טו) וַיִּתְאַוֶּה דָוִד וַיֹּאמַר מִי יַשְׁקֵנִי מַיִם מִבֹּאר בֵּית לֶחֶם אֲשֶׁר בַּשָּׁעַר: (טז) וַיִּבְקְעוּ שְׁלֹשֶׁת הַגִּבֹּרִים בְּמַחֲנֵה פְלִשְׁתִּים וַיִּשְׁאֲבוּ מַיִם מִבֹּאר בֵּית לֶחֶם אֲשֶׁר בַּשַּׁעַר וַיִּשְׂאוּ וַיָּבִאוּ אֶל דָּוִד וְלֹא אָבָה לִשְׁתּוֹתָם וַיַּסֵּךְ אֹתָם לַיקֹוָק: (יז) וַיֹּאמֶר חָלִילָה לִּי יְקֹוָק מֵעֲשֹׂתִי זֹאת הֲדַם הָאֲנָשִׁים הַהֹלְכִים בְּנַפְשׁוֹתָם וְלֹא אָבָה לִשְׁתּוֹתָם אֵלֶּה עָשׂוּ שְׁלֹשֶׁת הַגִּבֹּרִים: ס (יח) וְהִכָּה אֶת האריה האריה הָאֲרִי הָאֲרִי בְּתוֹךְ הַבֹּאר בְּיוֹם הַשָּׁלֶג: (כא) וְהוּא הִכָּה אֶת אִישׁ מִצְרִי אשר אשר אִישׁ אִישׁ מַרְאֶה וּבְיַד הַמִּצְרִי חֲנִית וַיֵּרֶד אֵלָיו בַּשָּׁבֶט וַיִּגְזֹל אֶת הַחֲנִית מִיַּד הַמִּצְרִי וַיַּהַרְגֵהוּ בַּחֲנִיתוֹ: (כב) אֵלֶּה עָשָׂה בְּנָיָהוּ בֶּן יְהוֹיָדָע וְלוֹ שֵׁם בִּשְׁלֹשָׁה הַגִּבֹּרִים: (כג) מִן הַשְּׁלֹשִׁים נִכְבָּד וְאֶל
