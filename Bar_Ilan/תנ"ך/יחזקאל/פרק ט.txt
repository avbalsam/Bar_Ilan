
 יחזקאל פרק ט (א) וַיִּקְרָא בְאָזְנַי קוֹל גָּדוֹל לֵאמֹר קָרְבוּ פְּקֻדּוֹת הָעִיר וְאִישׁ כְּלִי מַשְׁחֵתוֹ בְּיָדוֹ: (ב) וְהִנֵּה שִׁשָּׁה אֲנָשִׁים בָּאִים מִדֶּרֶךְ שַׁעַר הָעֶלְיוֹן אֲשֶׁר מָפְנֶה צָפוֹנָה וְאִישׁ כְּלִי מַפָּצוֹ בְּיָדוֹ וְאִישׁ אֶחָד בְּתוֹכָם לָבֻשׁ בַּדִּים וְקֶסֶת הַסֹּפֵר בְּמָתְנָיו וַיָּבֹאוּ וַיַּעַמְדוּ אֵצֶל מִזְבַּח הַנְּחֹשֶׁת: (ג) וּכְבוֹד אֱלֹהֵי יִשְׂרָאֵל נַעֲלָה מֵעַל הַכְּרוּב אֲשֶׁר הָיָה עָלָיו אֶל מִפְתַּן הַבָּיִת וַיִּקְרָא אֶל הָאִישׁ הַלָּבֻשׁ הַבַּדִּים אֲשֶׁר קֶסֶת הַסֹּפֵר בְּמָתְנָיו: ס (ד) וַיֹּאמֶר יְקֹוָק אלו אלו אֵלָיו אֵלָיו עֲבֹר בְּתוֹךְ הָעִיר בְּתוֹךְ יְרוּשָׁלִָם וְהִתְוִיתָ תָּו עַל מִצְחוֹת הָאֲנָשִׁים הַנֶּאֱנָחִים וְהַנֶּאֱנָקִים עַל כָּל הַתּוֹעֵבוֹת הַנַּעֲשׂוֹת בְּתוֹכָהּ: (ה) וּלְאֵלֶּה אָמַר בְּאָזְנַי עִבְרוּ בָעִיר אַחֲרָיו וְהַכּוּ על על אַל אַל תָּחֹס עיניכם עיניכם עֵינְכֶם עֵינְכֶם וְאַל תַּחְמֹלוּ: (ו) זָקֵן בָּחוּר וּבְתוּלָה וְטַף וְנָשִׁים תַּהַרְגוּ לְמַשְׁחִית וְעַל כָּל אִישׁ אֲשֶׁר עָלָיו הַתָּו אַל תִּגַּשׁוּ וּמִמִּקְדָּשִׁי תָּחֵלּוּ וַיָּחֵלּוּ בָּאֲנָשִׁים הַזְּקֵנִים אֲשֶׁר לִפְנֵי הַבָּיִת: (ז) וַיֹּאמֶר אֲלֵיהֶם טַמְּאוּ אֶת הַבַּיִת וּמַלְאוּ אֶת הַחֲצֵרוֹת חֲלָלִים צֵאוּ וְיָצְאוּ וְהִכּוּ בָעִיר: (ח) וַיְהִי כְּהַכּוֹתָם וְנֵאשֲׁאַר אָנִי וָאֶפְּלָה עַל פָּנַי וָאֶזְעַק וָאֹמַר אֲהָהּ אֲדֹנָי יְקֹוִק הֲמַשְׁחִית אַתָּה אֵת כָּל שְׁאֵרִית יִשְׂרָאֵל בְּשָׁפְכְּךָ אֶת חֲמָתְךָ עַל יְרוּשָׁלִָם: (ט) וַיֹּאמֶר אֵלַי עֲוֹן בֵּית יִשְׂרָאֵל וִיהוּדָה גָּדוֹל בִּמְאֹד מְאֹד וַתִּמָּלֵא הָאָרֶץ דָּמִים וְהָעִיר מָלְאָה מֻטֶּה כִּי אָמְרוּ עָזַב יְקֹוָק אֶת הָאָרֶץ וְאֵין יְקֹוָק רֹאֶה: (י) וְגַם אֲנִי לֹא תָחוֹס עֵינִי וְלֹא אֶחְמֹל דַּרְכָּם בְּרֹאשָׁם נָתָתִּי: (יא) וְהִנֵּה הָאִישׁ לְבֻשׁ הַבַּדִּים אֲשֶׁר הַקֶּסֶת בְּמָתְנָיו מֵשִׁיב דָּבָר לֵאמֹר עָשִׂיתִי כאשר כאשר כְּכֹל כְּכֹל אֲשֶׁר אֲשֶׁר צִוִּיתָנִי: ס
