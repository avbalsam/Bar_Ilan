
 משלי פרק כג (א) כִּי תֵשֵׁב לִלְחוֹם אֶת מוֹשֵׁל בִּין תָּבִין אֶת אֲשֶׁר לְפָנֶיךָ: (ב) וְשַׂמְתָּ שַׂכִּין בְּלֹעֶךָ אִם בַּעַל נֶפֶשׁ אָתָּה: (ג) אַל תִּתְאָו לְמַטְעַמּוֹתָיו וְהוּא לֶחֶם כְּזָבִים: (ד) אַל תִּיגַע לְהַעֲשִׁיר מִבִּינָתְךָ חֲדָל: (ה) התעוף התעוף הֲתָעִיף הֲתָעִיף עֵינֶיךָ בּוֹ וְאֵינֶנּוּ כִּי עָשֹׂה יַעֲשֶׂה לּוֹ כְנָפַיִם כְּנֶשֶׁר ועיף ועיף יָעוּף יָעוּף הַשָּׁמָיִם: פ (ו) אַל תִּלְחַם אֶת לֶחֶם רַע עָיִן וְאַל תתאו תתאו תִּתְאָיו תִּתְאָיו לְמַטְעַמֹּתָיו: (ז) כִּי כְּמוֹ שָׁעַר בְּנַפְשׁוֹ כֶּן הוּא אֱכֹל וּשְׁתֵה יֹאמַר לָךְ וְלִבּוֹ בַּל עִמָּךְ: (ח) פִּתְּךָ אָכַלְתָּ תְקִיאֶנָּה וְשִׁחַתָּ דְּבָרֶיךָ הַנְּעִימִים: (ט) בְּאָזְנֵי כְסִיל אַל תְּדַבֵּר כִּי יָבוּז לְשֵׂכֶל מִלֶּיךָ: (י) אַל תַּסֵּג גְּבוּל עוֹלָם וּבִשְׂדֵי יְתוֹמִים אַל תָּבֹא: (יא) כִּי גֹאֲלָם חָזָק הוּא יָרִיב אֶת רִיבָם אִתָּךְ: (יב) הָבִיאָה לַמּוּסָר לִבֶּךָ וְאָזְנֶךָ לְאִמְרֵי דָעַת: (יג) אַל תִּמְנַע מִנַּעַר מוּסָר כִּי תַכֶּנּוּ בַשֵּׁבֶט לֹא יָמוּת: (יד) אַתָּה בַּשֵּׁבֶט תַּכֶּנּוּ וְנַפְשׁוֹ מִשְּׁאוֹל תַּצִּיל: (טו) בְּנִי אִם חָכַם לִבֶּךָ יִשְׂמַח לִבִּי גַם אָנִי: (טז) וְתַעְלֹזְנָה כִלְיוֹתָי בְּדַבֵּר שְׂפָתֶיךָ מֵישָׁרִים: (יז) אַל יְקַנֵּא לִבְּךָ בַּחַטָּאִים כִּי אִם בְּיִרְאַת יְקֹוָק כָּל הַיּוֹם: (יח) כִּי אִם יֵשׁ אַחֲרִית וְתִקְוָתְךָ לֹא תִכָּרֵת: (יט) שְׁמַע אַתָּה בְנִי וַחֲכָם וְאַשֵּׁר בַּדֶּרֶךְ לִבֶּךָ: (כ) אַל תְּהִי בְסֹבְאֵי יָיִן בְּזֹלֲלֵי בָשָׂר לָמוֹ: (כא) כִּי סֹבֵא וְזוֹלֵל יִוָּרֵשׁ וּקְרָעִים תַּלְבִּישׁ נוּמָה: (כב) שְׁמַע לְאָבִיךָ זֶה יְלָדֶךָ וְאַל תָּבוּז כִּי זָקְנָה אִמֶּךָ: (כג) אֱמֶת קְנֵה וְאַל תִּמְכֹּר חָכְמָה וּמוּסָר וּבִינָה: (כד) גול גול גִּיל גִּיל יגול יגול יָגִיל יָגִיל אֲבִי צַדִּיק יולד יולד וְיוֹלֵד וְיוֹלֵד חָכָם וישמח וישמח יִשְׂמַח יִשְׂמַח בּוֹ: (כה) יִשְׂמַח אָבִיךָ וְאִמֶּךָ וְתָגֵל יוֹלַדְתֶּךָ: (כו) תְּנָה בְנִי לִבְּךָ לִי וְעֵינֶיךָ דְּרָכַי תרצנה תרצנה תִּצֹּרְנָה תִּצֹּרְנָה : (כז) כִּי שׁוּחָה עֲמֻקָּה זוֹנָה וּבְאֵר צָרָה נָכְרִיָּה: (כח) אַף הִיא כְּחֶתֶף תֶּאֱרֹב וּבוֹגְדִים בְּאָדָם תּוֹסִף: (כט) לְמִי אוֹי לְמִי אֲבוֹי לְמִי מדונים מדונים מִדְיָנִים מִדְיָנִים לְמִי שִׂיחַ לְמִי פְּצָעִים חִנָּם לְמִי חַכְלִלוּת עֵינָיִם: (ל) לַמְאַחֲרִים עַל הַיָּיִן לַבָּאִים לַחְקֹר מִמְסָךְ: (לא) אַל תֵּרֶא יַיִן כִּי יִתְאַדָּם כִּי יִתֵּן בכיס בכיס בַּכּוֹס בַּכּוֹס עֵינוֹ יִתְהַלֵּךְ בְּמֵישָׁרִים: (לב) אַחֲרִיתוֹ כְּנָחָשׁ יִשָּׁךְ וּכְצִפְעֹנִי יַפְרִשׁ: (לג) עֵינֶיךָ יִרְאוּ זָרוֹת וְלִבְּךָ יְדַבֵּר תַּהְפֻּכוֹת: (לד) וְהָיִיתָ כְּשֹׁכֵב בְּלֶב יָם וּכְשֹׁכֵב בְּרֹאשׁ חִבֵּל: (לה) הִכּוּנִי בַל חָלִיתִי הֲלָמוּנִי בַּל יָדָעְתִּי מָתַי אָקִיץ אוֹסִיף אֲבַקְשֶׁנּוּ עוֹד:
